#find direct train function
def find_trains(source, destination):

    results = []

    for train_no, group in df.groupby('Train_No'):

        stations = list(group['Station_Name'])

        if source in stations and destination in stations:

            source_idx = stations.index(source)
            dest_idx = stations.index(destination)

            if source_idx < dest_idx:

                train_no = group[
                    'Train_Name'
                ].iloc[0]

                results.append(
                    [train_no, train_name]
                )

    return results

#calculate journey duration
def get_duration(train_no,
                 source,
                 destination):

    train = df[
        df['Train_No'] == train_no
    ]

    source_row = train[
        train['Station_Name'] == source
    ].iloc[0]

    dest_row = train[
        train['Station_Name'] == destination
    ].iloc[0]

    start = pd.to_datetime(
        source_row['departure_time']
    )

    end = pd.to_datetime(
        dest_row['Arrival_Time']
    )

    duration = end - start

    return duration
